# For Docker (Debian/Ubuntu-based images):
#   apt-get install -y libpango-1.0-0 libpangocairo-1.0-0 libpangoft2-1.0-0 \
#     libharfbuzz0b libfribidi0 libgdk-pixbuf2.0-0 libffi8 libcairo2 \
#     libgirepository-1.0-1 gir1.2-pango-1.0 shared-mime-info \
#     fonts-liberation fonts-dejavu-core
#
# For Ubuntu/Debian servers:
#   sudo apt-get install libpango-1.0-0 libpangocairo-1.0-0 libgdk-pixbuf2.0-0 \
#     libffi-dev shared-mime-info
#
# For macOS:
#   brew install pango libffi cairo gdk-pixbuf
#
# For Alpine Linux:
#   apk add pango cairo gdk-pixbuf font-noto
#
# See: https://doc.courtbouillon.org/weasyprint/stable/first_steps.html
